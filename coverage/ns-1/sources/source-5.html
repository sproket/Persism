333     static synchronized <T> Collection<PropertyInfo> determinePropertyInfo(Class<T> objectClass) {
334         if (propertyMap.containsKey(objectClass)) {
335             return propertyMap.get(objectClass);
336         }
337 
338         Map<String, PropertyInfo> propertyNames = new HashMap<>(32);
339 
340         List<Field> fields = new ArrayList<>(32);
341 
342         // getDeclaredFields does not get fields from super classes.....
343         fields.addAll(Arrays.asList(objectClass.getDeclaredFields()));
344         Class<?> sup = objectClass.getSuperclass();
345         log.debug("fields for %s", sup);
346         while (!sup.equals(Object.class) && !sup.equals(PersistableObject.class)) {
347             fields.addAll(Arrays.asList(sup.getDeclaredFields()));
348             sup = sup.getSuperclass();
349             log.debug("fields for %s", sup);
350         }
351 
352         Method[] methods = objectClass.getMethods();
353 
354         for (Field field : fields) {
355             log.debug("Field Name: %s", field.getName());
356             String propertyName = field.getName().substring(0, 1).toUpperCase() + field.getName().substring(1);
357             log.debug("Property Name: *%s* ", propertyName);
358 
359             PropertyInfo propertyInfo = new PropertyInfo();
360             propertyInfo.propertyName = propertyName;
361             propertyInfo.field = field;
362             Annotation[] annotations = field.getAnnotations();
363             for (Annotation annotation : annotations) {
364                 propertyInfo.annotations.put(annotation.annotationType(), annotation);
365             }
366 
367             for (Method method : methods) {
368                 String propertyNameToTest = propertyName;
369                 if (propertyNameToTest.startsWith("Is") && propertyNameToTest.length() > 2 && Character.isUpperCase(propertyNameToTest.charAt(2))) {
370                     propertyNameToTest = propertyName.substring(2);
371                 }
372 
373                 String[] candidates = {"set" + propertyNameToTest, "get" + propertyNameToTest, "is" + propertyNameToTest, field.getName()};
374 
375                 if (Arrays.asList(candidates).contains(method.getName())) {
376                     log.debug("  METHOD: %s", method.getName());
377 
378                     annotations = method.getAnnotations();
379                     for (Annotation annotation : annotations) {
380                         propertyInfo.annotations.put(annotation.annotationType(), annotation);
381                     }
382 
383                     if (method.getName().equalsIgnoreCase("set" + propertyNameToTest)) {
384                         propertyInfo.setter = method;
385                     } else {
386                         propertyInfo.getter = method;
387                     }
388                 }
389             }
390 
391             propertyInfo.readOnly = propertyInfo.setter == null;
392             propertyNames.put(propertyName.toLowerCase(), propertyInfo);
393         }
394 
395         // Remove any properties found with the NotColumn annotation
396         // http://stackoverflow.com/questions/2026104/hashmap-keyset-foreach-and-remove
397         Iterator<Map.Entry<String, PropertyInfo>> it = propertyNames.entrySet().iterator();
398         while (it.hasNext()) {
399             Map.Entry<String, PropertyInfo> entry = it.next();
400             PropertyInfo info = entry.getValue();
401             if (info.getAnnotation(NotColumn.class) != null) {
402                 it.remove();
403             }
404         }
405 
406         Collection<PropertyInfo> properties = Collections.unmodifiableCollection(propertyNames.values());
407         propertyMap.put(objectClass, properties);
408         return properties;
409     }

693

701                 return columns;

791     // Returns the table name found in the DB in the same case as in the DB.
792     // throws PersismException if we cannot guess any table name for this class.
793     private <T> String guessTableName(Class<T> objectClass) throws PersismException {
794         Set<String> guesses = new LinkedHashSet<>(6); // guess order is important
795         List<String> guessedTables = new ArrayList<String>(6);
796 
797         String className = objectClass.getSimpleName();
798 
799         addTableGuesses(className, guesses);
800         for (String tableName : tableNames) {
801             for (String guess : guesses) {
802                 if (guess.equalsIgnoreCase(tableName)) {
803                     guessedTables.add(tableName);
804                 }
805             }
806         }
807         if (guessedTables.size() == 0) {
808             throw new PersismException("Could not determine a table for type: " + objectClass.getName() + " Guesses were: " + guesses);
809         }
810 
811         if (guessedTables.size() > 1) {
812             throw new PersismException("Could not determine a table for type: " + objectClass.getName() + " Guesses were: " + guesses + " and we found multiple matching tables: " + guessedTables);
813         }
814         return guessedTables.get(0);
815     }
