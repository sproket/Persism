378     private static synchronized <T> Collection<PropertyInfo> determinePropertyInfo(Class<T> objectClass) {
379         if (propertyMap.containsKey(objectClass)) {
380             return propertyMap.get(objectClass);
381         }
382 
383         Map<String, PropertyInfo> propertyInfos = new HashMap<>(32);
384 
385         List<Field> fields = new ArrayList<>(32);
386 
387         // getDeclaredFields does not get fields from super classes.....
388         fields.addAll(Arrays.asList(objectClass.getDeclaredFields()));
389         Class<?> sup = objectClass.getSuperclass();
390         log.debug("fields for %s", sup);
391         while (!sup.equals(Object.class) && !sup.equals(PersistableObject.class)) {
392             fields.addAll(Arrays.asList(sup.getDeclaredFields()));
393             sup = sup.getSuperclass();
394             log.debug("fields for %s", sup);
395         }
396 
397         Method[] methods = objectClass.getMethods();
398 
399         for (Field field : fields) {
400             // Skip static fields
401             if (Modifier.isStatic(field.getModifiers())) {
402                 continue;
403             }
404 //            log.debug("Field Name: %s", field.getName());
405             String propertyName = field.getName();
406 //            log.debug("Property Name: *%s* ", propertyName);
407 
408             PropertyInfo propertyInfo = new PropertyInfo();
409             propertyInfo.propertyName = propertyName;
410             propertyInfo.field = field;
411             Annotation[] annotations = field.getAnnotations();
412             for (Annotation annotation : annotations) {
413                 propertyInfo.annotations.put(annotation.annotationType(), annotation);
414             }
415 
416             for (Method method : methods) {
417                 String propertyNameToTest = field.getName().substring(0, 1).toUpperCase() + field.getName().substring(1);
418                 // log.debug("property name for testing %s", propertyNameToTest);
419                 if (propertyNameToTest.startsWith("Is") && propertyNameToTest.length() > 2 && Character.isUpperCase(propertyNameToTest.charAt(2))) {
420                     propertyNameToTest = propertyName.substring(2);
421                 }
422 
423                 String[] candidates = {"set" + propertyNameToTest, "get" + propertyNameToTest, "is" + propertyNameToTest, field.getName()};
424 
425                 if (Arrays.asList(candidates).contains(method.getName())) {
426                     //log.debug("  METHOD: %s", method.getName());
427 
428                     annotations = method.getAnnotations();
429                     for (Annotation annotation : annotations) {
430                         propertyInfo.annotations.put(annotation.annotationType(), annotation);
431                     }
432 
433                     // OR added to fix to builder pattern style when your setters are just the field name
434                     if (method.getName().equalsIgnoreCase("set" + propertyNameToTest) || method.getParameterCount() > 0) {
435                         propertyInfo.setter = method;
436                     } else {
437                         propertyInfo.getter = method;
438                     }
439                 }
440             }
441 
442             propertyInfo.readOnly = propertyInfo.setter == null;
443             propertyInfo.isJoin = propertyInfo.getAnnotation(Join.class) != null;
444             propertyInfos.put(propertyName.toLowerCase(), propertyInfo);
445         }
446 
447         // Remove any properties found with the NotColumn annotation
448         // http://stackoverflow.com/questions/2026104/hashmap-keyset-foreach-and-remove
449         Iterator<Map.Entry<String, PropertyInfo>> it = propertyInfos.entrySet().iterator();
450         while (it.hasNext()) {
451             Map.Entry<String, PropertyInfo> entry = it.next();
452             PropertyInfo info = entry.getValue();
453 			// added support for transient
454             if (info.getAnnotation(NotColumn.class) != null || Modifier.isTransient(info.field.getModifiers())) {
455                 it.remove();
456             }
457         }
458 
459         Collection<PropertyInfo> properties = Collections.unmodifiableCollection(propertyInfos.values());
460         propertyMap.put(objectClass, properties);
461 
462         // If a view or query - warn if we find any setters
463         if (objectClass.getAnnotation(NotTable.class) != null || objectClass.getAnnotation(View.class) != null) {
464             List<String> setters = new ArrayList<>();
465             for (PropertyInfo propertyInfo : properties) {
466                 if (propertyInfo.setter != null) {
467                     setters.add(propertyInfo.propertyName);
468                 }
469             }
470 
471             if (setters.size() > 0) {
472                 log.warn(Messages.SettersFoundInReadOnlyObject.message(objectClass, setters));
473             }
474         }
475 
476         return properties;
477     }

910 
911         if (tableOrViewMap.containsKey(objectClass)) {
912             return tableOrViewMap.get(objectClass);
913         }
914 
915         String tableName;
916         Table tableAnnotation = objectClass.getAnnotation(Table.class);
917         View viewAnnotation = objectClass.getAnnotation(View.class);
918 
919         if (tableAnnotation != null) {
920             tableName = tableAnnotation.value();
921             // double check against stored table names to get the actual case of the name
922             boolean found = false;
923             for (String name : tableNames) {
924                 if (name.equalsIgnoreCase(tableName)) {
925                     tableName = name;
926                     found = true;
927                 }
928             }
929             if (!found) {
930                 throw new PersismException(Messages.CouldNotFindTableNameInTheDatabase.message(tableName, objectClass.getName()));
931             }
932         } else if (viewAnnotation != null && isNotEmpty(viewAnnotation.value())) {
933 
934             tableName = viewAnnotation.value();
935 
936             // double check against stored view names to get the actual case of the name
937             boolean found = false;
938             for (String name : viewNames) {
939                 if (name.equalsIgnoreCase(tableName)) {
940                     tableName = name;
941                     found = true;
942                 }
943             }
944             if (!found) {
945                 throw new PersismException(Messages.CouldNotFindViewNameInTheDatabase.message(tableName, objectClass.getName()));
946             }
947         } else {

954     // Returns the table/view name found in the DB in the same case as in the DB.
955     // throws PersismException if we cannot guess any table/view name for this class.
956     private <T> String guessTableOrViewName(Class<T> objectClass) throws PersismException {
957         Set<String> guesses = new LinkedHashSet<>(6); // guess order is important
958         List<String> guessedTables = new ArrayList<>(6);
959 
960         String className = objectClass.getSimpleName();
961 
962         Set<String> list;
963         boolean isView = false;
964         if (objectClass.getAnnotation(View.class) != null) {
965             list = viewNames;
966             isView = true;
967         } else {
968             list = tableNames;
969         }
970 
971         addTableGuesses(className, guesses);
972         for (String tableName : list) {
973             for (String guess : guesses) {
974                 if (guess.equalsIgnoreCase(tableName)) {
975                     guessedTables.add(tableName);
976                 }
977             }
978         }
979         if (guessedTables.size() == 0) {
980             throw new PersismException(Messages.CouldNotDetermineTableOrViewForType.message(isView ? "view" : "table", objectClass.getName(), guesses));
981         }
982 
983         if (guessedTables.size() > 1) {
984             throw new PersismException(Messages.CouldNotDetermineTableOrViewForTypeMultipleMatches.message(isView ? "view" : "table", objectClass.getName(), guesses, guessedTables));
985         }
986         return guessedTables.get(0);
987     }
