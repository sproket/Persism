191

199                 propertyNames.clear();

202                 break;
203             }

206             ConstructorProperties constructorProperties = constructor.getAnnotation(ConstructorProperties.class);
207             if (constructorProperties != null) {

209                 if (listEqualsIgnoreOrder(propertyNames, parameterNames)) {
