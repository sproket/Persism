612

616             }
617 
618             // https://forums.oracle.com/forums/thread.jspa?threadID=879222
619             // http://download.oracle.com/javase/1.4.2/docs/guide/jdbc/getstart/statement.html
620             //int ret = st.executeUpdate(insertStatement, Statement.RETURN_GENERATED_KEYS);
621             assert params.size() == columnInfos.size();
622             for (int j = 0; j < params.size(); j++) {
623                 if (params.get(j) != null) {
624                     params.set(j, converter.convert(params.get(j), columnInfos.get(j).columnType.getJavaType(), columnInfos.get(j).columnName));
625                 }
626             }
627 
628             helper.setParameters(st, params.toArray());
629             st.execute();
630             int ret = st.getUpdateCount();
631 
632             log.debug("insert return count after insert: %s", ret);
633 
634             List<Object> primaryKeyValues = new ArrayList<>();
635             if (generatedKeys.size() > 0) {
636                 rs = st.getGeneratedKeys();
637                 PropertyInfo propertyInfo;
638                 for (String column : generatedKeys) {
639                     if (rs.next()) {
640

642 
643                         Method setter = propertyInfo.setter;
644                         Object value;
645                         if (setter != null) {
646                             value = helper.getTypedValueReturnedFromGeneratedKeys(setter.getParameterTypes()[0], rs);
647                             setter.invoke(object, value);
648                         } else {
649                             // Set read-only property by field ONLY FOR NON-RECORDS.
650                             value = helper.getTypedValueReturnedFromGeneratedKeys(propertyInfo.field.getType(), rs);
651                             if (!isRecord(object.getClass())) {
652                                 propertyInfo.field.setAccessible(true);
653                                 propertyInfo.field.set(object, value);
654                                 propertyInfo.field.setAccessible(false);
655                                 log.debug("insert %s generated %s", column, value);
656                             }
657                         }
658 
659                         primaryKeyValues.add(value);
660                     }
661                 }
662             }
663 
664             // If it's a record we can't assign the autoinc so we need a refresh
665             if (generatedKeys.size() > 0 && isRecord(object.getClass())) {
666                 refreshAfterInsert = true;
667             }
668 
669             Object returnObject = null;
670             if (refreshAfterInsert) {
671                 // these 2 fetches need a fetchAfterInsert flag
672                 // Read the full object back to update any properties which had defaults
673                 if (isRecord(object.getClass())) {
674                     SQL sql = new SQL(metaData.getDefaultSelectStatement(object.getClass(), connection));
675                     returnObject = fetch(object.getClass(), sql, params(primaryKeyValues.toArray()));
676                 } else {
677                     fetch(object);
678                     returnObject = object;
679                 }
680             } else {
681                 returnObject = object;
682             }
683 
684             if (object instanceof Persistable<?> pojo) {
685                 // Save this object new state to later detect changed properties
686                 pojo.saveReadState();
687             }
688 
689             //noinspection unchecked
690             return new Result<>(ret, (T) returnObject);
691         } catch (Exception e) {
692             Util.rollback(connection);
693             throw new PersismException(e.getMessage(), e);
694         } finally {
695             Util.cleanup(st, rs);
696         }
697     }

752      * This method will set autocommit to false then execute the function, commit and set autocommit back to true.
753      * <pre>{@code

764      * });
765      * }</pre>
766      *
767      * @param transactionBlock Block of operations expected to run as a single transaction.
768      * @throws PersismException in case of SQLException where the transaction is rolled back.
769      */
770     public void withTransaction(Runnable transactionBlock) {
771         try {
772             connection.setAutoCommit(false);
773             transactionBlock.run();
774             connection.commit();
775         } catch (Exception e) {
776             Util.rollback(connection);
777             throw new PersismException(e.getMessage(), e);
778         } finally {
779             try {
780                 connection.setAutoCommit(true);
781             } catch (SQLException e) {
782                 log.warn(e.getMessage());
783             }
784         }
785     }
